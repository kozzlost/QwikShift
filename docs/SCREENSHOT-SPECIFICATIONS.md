                                                                                                                                                                                                           ## Design Guidelines
                                                                                                                                                                                                           
                                                                                                                                                                                                           ### Color Palette
                                                                                                                                                                                                           - Primary Blue: #0066FF
                                                                                                                                                                                                           - - Dark Gray: #333333
                                                                                                                                                                                                             - - Light Gray: #F5F5F5
                                                                                                                                                                                                               - - Accent Gold: #FFB800 (for badges)
                                                                                                                                                                                                                 - - Success Green: #28A745
                                                                                                                                                                                                                   - - White: #FFFFFF
                                                                                                                                                                                                                    
                                                                                                                                                                                                                     - ### Typography
                                                                                                                                                                                                                     - - Headlines: Bold, 20-28px
                                                                                                                                                                                                                       - - Body: Regular, 14-16px
                                                                                                                                                                                                                         - - Captions: Light, 12-14px
                                                                                                                                                                                                                           - - Font: System font (San Francisco / SF Pro)
                                                                                                                                                                                                                            
                                                                                                                                                                                                                             - ### Safe Areas
                                                                                                                                                                                                                             - - Top: 44px (status bar + notch)
                                                                                                                                                                                                                               - - Bottom: 34px (home indicator)
                                                                                                                                                                                                                                 - - Sides: 16px padding
                                                                                                                                                                                                                                  
                                                                                                                                                                                                                                   - ### Image Requirements
                                                                                                                                                                                                                                   - - Format: PNG with transparency (if needed)
                                                                                                                                                                                                                                     - - Resolution: Exact device dimensions
                                                                                                                                                                                                                                       - - Compression: Optimized for App Store (< 10MB per set)
                                                                                                                                                                                                                                         - - No watermarks or branding overlays outside safe areas
                                                                                                                                                                                                                                          
                                                                                                                                                                                                                                           - ### Text Overlay Specifications
                                                                                                                                                                                                                                           - All screenshots should include:
                                                                                                                                                                                                                                           - - Title text in white, bold, 18-24px
                                                                                                                                                                                                                                             - - Subtitle text in white, regular, 14-16px
                                                                                                                                                                                                                                               - - Positioned at bottom
                                                                                                                                                                                                                                                 - - Dark semi-transparent background bar (rgba(0,0,0,0.6))
                                                                                                                                                                                                                                                   - - 20px padding from bottom safe area
